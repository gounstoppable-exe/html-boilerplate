HTML Boilerplate Template.

This is code of a advanced html boilerplate with meta tags and other details. Meta tags helps to enhance the SEO Score of the website.

                
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Website Title Here</title>
    <!-- ====== CSS FILE LINKS ======  -->
    <link rel="stylesheet" href="style1.css">
    <link rel="stylesheet" href="style2.css">
    <!-- ====== ADDITIONAL CDN LINKS ======  -->
    <!-- **Link google fonts, or any cdn link here** -->
    <!-- ====== META TAGS FOR SEO ======  -->

    <!-- Primary Meta Tags -->
    <meta name="title" content="Title of your website.">
    <meta name="description" content="Write a description of your website here. Write it briefly, clean & make sure it resembles what your site is about.">
    <!-- Open Graph / Facebook -->
    <meta property="og:type" content="website">
    <meta property="og:url" content="https://linkofyourwebsite.com/">
    <meta property="og:title" content="Title of your website.">
    <meta property="og:description" content="Write a description of your website here. Write it briefly, clean & make sure it resembles what your site is about.">
    <meta property="og:image" content="https://linkofyourwebsite.com/assets/yourImageUrl.jpg">
    <!-- Twitter -->
    <meta property="twitter:card" content="summary_large_image">
    <meta property="twitter:url" content="https://linkofyourwebsite.com/">
    <meta property="twitter:title" content="Title of your website.">
    <meta property="twitter:description" content="Write a description of your website here. Write it briefly, clean & make sure it resembles what your site is about.">
    <meta property="twitter:image" content="https://linkofyourwebsite.com/assets/yourImageUrl.jpg">
</head>
<body>

    <!-- ====== MAIN HTML CODE ====== -->
    


    <!-- ====== SCRIPT FILES ======  -->
    <script src="scriptfile1.js"></script>
    <script src="scriptfile2.js"></script>
</body>
</html>
